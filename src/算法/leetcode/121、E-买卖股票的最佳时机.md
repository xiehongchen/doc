/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
  let i = 0
  let min = prices[0]
  for (const j of prices) {
    i = Math.max(i, j - min)
    min = Math.min(min, j)
  }
  return i
};
